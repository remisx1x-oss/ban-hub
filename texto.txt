    let isOpen = false;

    function toggleHub() {
        if (isOpen) {
            closeHub();
        } else {
            openHub();
        }
    }

    function openHub() {
        const container = document.getElementById('hubContainer');
        const overlay = document.getElementById('overlay');
        
        container.classList.add('show');
        overlay.classList.add('show');
        isOpen = true;
    }

    function closeHub() {
        const container = document.getElementById('hubContainer');
        const overlay = document.getElementById('overlay');
        
        container.classList.remove('show');
        overlay.classList.remove('show');
        isOpen = false;
    }

    // Funci√≥n para activar Fly B
    function activateFlyB() {
        try {
            showNotification('‚úàÔ∏è Cargando Fly V3...', 'info');
            
            const flyScript = `loadstring(game:HttpGet("https://rawscripts.net/raw/Universal-Script-Fly-V3-48359"))()`;
            
            navigator.clipboard.writeText(flyScript).then(() => {
                showNotification('üìã Script copiado al portapapeles!', 'success');
            }).catch(() => {
                showNotification('‚ùå Error al copiar script', 'error');
            });
            
            const button = document.querySelector('.script-button');
            const originalText = button.innerHTML;
            button.innerHTML = '‚úÖ LISTO';
            button.style.background = 'linear-gradient(45deg, #00ff88, #00cc6a)';
            
            console.log('Fly V3 Script:', flyScript);
            
            setTimeout(() => {
                button.innerHTML = originalText;
                button.style.background = 'linear-gradient(45deg, #00ff88, #00cc6a)';
            }, 2000);
            
        } catch (error) {
            showNotification('‚ùå Error al cargar Fly V3', 'error');
            console.error('Error:', error);
        }
    }

    // Funci√≥n para mostrar notificaciones
    function showNotification(message, type = 'info') {
        const notification = document.createElement('div');
        let bgColor;
        
        switch(type) {
            case 'success':
                bgColor = 'linear-gradient(45deg, #00ff88, #00cc6a)';
                break;
            case 'error':
                bgColor = 'linear-gradient(45deg, #ff0080, #cc0066)';
                break;
            default:
                bgColor = 'linear-gradient(45deg, #8000ff, #6600cc)';
        }
        
        notification.style.cssText = `
            position: fixed;
            top: 100px;
            left: 20px;
            background: ${bgColor};
            color: white;
            padding: 15px 25px;
            font-weight: 700;
            font-family: 'Orbitron', monospace;
            text-transform: uppercase;
            letter-spacing: 1px;
            font-size: 12px;
            z-index: 1002;
            transform: translateX(-400px);
            transition: transform 0.4s ease;
            box-shadow: 0 0 20px rgba(0, 0, 0, 0.5);
            clip-path: polygon(0 0, calc(100% - 15px) 0, 100% 15px, 100% 100%, 15px 100%, 0 calc(100% - 15px));
            border: 2px solid rgba(255, 255, 255, 0.2);
        `;
        notification.textContent = message;
        document.body.appendChild(notification);
        
        setTimeout(() => {
            notification.style.transform = 'translateX(0)';
        }, 100);
        
        setTimeout(() => {
            notification.style.transform = 'translateX(-400px)';
            setTimeout(() => {
                if (document.body.contains(notification)) {
                    document.body.removeChild(notification);
                }
            }, 400);
        }, 3000);
    }

    // Cerrar con tecla ESC
    document.addEventListener('keydown', function(event) {
        if (event.key === 'Escape' && isOpen) {
            closeHub();
        }
    });
</script>
